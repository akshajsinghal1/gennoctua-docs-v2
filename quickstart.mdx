## Fastest possible setup

Add this to any HTML page and change the 3 values:

```html
<script src="https://cdn.jsdelivr.net/gh/akshajsinghal1/gennoctua-sdk-v2@main/dist/cdn/personalize.min.global.js"></script>

<input id="photos" type="file" multiple accept="image/*" webkitdirectory />
<button id="tryOnBtn" disabled>âœ¨ Try On</button>
<p id="status"></p>
<img id="result" style="display:none; max-width:500px" />

<script>
  const BRAND_KEY = "gn_pk_yourbrand_xxx";        // â† your brand key
  const PRODUCT_IMAGE_URL = "https://...";          // â† product image URL
  const PRODUCT_TYPE = "sunglasses";                // â† product type

  let sdk;
  window.Personalize.init({ auth: { publicKey: BRAND_KEY } })
    .then(s => { sdk = s; });

  document.getElementById("photos").addEventListener("change", async (e) => {
    document.getElementById("status").textContent = "Reading your photos...";
    const summary = await sdk.ingestImages(e.target.files, p => {
      document.getElementById("status").textContent = p.message;
    });
    if (summary.availableCategories.length > 0) {
      document.getElementById("status").textContent = "âœ“ Ready! Click Try On.";
      document.getElementById("tryOnBtn").disabled = false;
    } else {
      document.getElementById("status").textContent = "No suitable photo found. Try different photos.";
    }
  });

  document.getElementById("tryOnBtn").addEventListener("click", async () => {
    document.getElementById("status").textContent = "Generating (~20 seconds)...";
    document.getElementById("tryOnBtn").disabled = true;
    try {
      const result = await sdk.personalize({
        imageUrl: PRODUCT_IMAGE_URL,
        productType: PRODUCT_TYPE,
        gender: "female"
      });
      document.getElementById("result").src = result.imageUrl;
      document.getElementById("result").style.display = "block";
      document.getElementById("status").textContent = "Here's how it looks on you!";
    } catch (err) {
      document.getElementById("status").textContent = "Something went wrong. Try again.";
      document.getElementById("tryOnBtn").disabled = false;
    }
  });
</script>
```

## What to change

<Steps>
  <Step title="Brand Key">
    Replace `gn_pk_yourbrand_xxx` with your brand key from Gennoctua.
  </Step>
  <Step title="Product Image URL">
    Replace with the URL of the product image on your page. Right-click the image â†’ Copy Image Address.
  </Step>
  <Step title="Product Type">
    Replace `"sunglasses"` with the correct type. See [Supported Products](/sdk/supported-products) for the full list.
  </Step>
</Steps>
