## Overview

In full-stack mode:
- Your **backend** holds the Hyperpersona API key (`HP_API_KEY`, issued by Gennoctua — never exposed to the browser)
- Your **frontend** uses the SDK — all requests go through your backend only
- The browser never calls Hyperpersona directly

```
Browser (SDK) → Your Backend → Hyperpersona (HP) API
```

```bash
npm install @gennoctua/personalize-core
```

## Step 2 — Backend Setup

### Store your keys

Add to your `.env` file:
```
HP_API_KEY=your_hyperpersona_api_key_here       # issued by Gennoctua — X-API-Key to Hyperpersona
JWT_SECRET=your_random_secret_here_min_32_chars # your own secret for signing session tokens
```

### Install backend dependencies

```bash
npm install jsonwebtoken
```

### Add token endpoint

This generates a short-lived JWT for each user session.

<CodeGroup>
```js Node/Express
const jwt = require("jsonwebtoken");

app.post("/auth/tryon-token", (req, res) => {
  const token = jwt.sign(
    { purpose: "tryon", iat: Date.now() },
    process.env.JWT_SECRET,
    { expiresIn: "1h" }
  );
  res.json({ token, expiresIn: 3600 });
});
```

```js Next.js API Route
// /pages/api/auth/tryon-token.js
import jwt from "jsonwebtoken";

export default function handler(req, res) {
  if (req.method !== "POST") return res.status(405).end();

  const token = jwt.sign(
    { purpose: "tryon" },
    process.env.JWT_SECRET,
    { expiresIn: "1h" }
  );
  res.json({ token, expiresIn: 3600 });
}
```
</CodeGroup>

### Add proxy endpoints

Your backend must proxy **all** SDK routes. The SDK calls **5** upstream Hyperpersona paths (plus your own `/auth/tryon-token`): try-on submit + SSE status, profile select, room select, and user-image upload. Forward each with `X-API-Key: process.env.HP_API_KEY`.

<CodeGroup>
```js Node/Express
const jwt = require("jsonwebtoken");
const HP_BASE = "https://hyperpersona-api-914497143112.us-central1.run.app";

function requireTryonToken(req, res, next) {
  const auth = req.headers.authorization || "";
  const token = auth.replace("Bearer ", "");
  try {
    jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    res.status(401).json({ error: "Invalid or expired token" });
  }
}

const hpHeadersJson = {
  "X-API-Key": process.env.HP_API_KEY,
  "Content-Type": "application/json",
};

// ── Try-on (person + furniture: same submit + SSE status on HP) ───────────

app.post("/api/tryon/submit", requireTryonToken, async (req, res) => {
  const response = await fetch(`${HP_BASE}/api/tryon/submit`, {
    method: "POST",
    headers: { "X-API-Key": process.env.HP_API_KEY },
    body: req,
  });
  const data = await response.json();
  res.json(data);
});

app.get("/api/tryon/status/:jobId", requireTryonToken, async (req, res) => {
  const response = await fetch(`${HP_BASE}/api/tryon/status/${req.params.jobId}`, {
    headers: { "X-API-Key": process.env.HP_API_KEY },
  });
  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");
  res.setHeader("Connection", "keep-alive");
  response.body.pipe(res);
});

// ── LLM refinement + upload (same HP host) ─────────────────────────────────

app.post("/api/tryon/api/profile/select", requireTryonToken, async (req, res) => {
  const response = await fetch(`${HP_BASE}/api/profile/select`, {
    method: "POST",
    headers: hpHeadersJson,
    body: JSON.stringify(req.body),
  });
  res.json(await response.json());
});

app.post("/api/tryon/api/room/select", requireTryonToken, async (req, res) => {
  const response = await fetch(`${HP_BASE}/api/room/select`, {
    method: "POST",
    headers: hpHeadersJson,
    body: JSON.stringify(req.body),
  });
  res.json(await response.json());
});

app.post("/api/tryon/api/uploads/user-image", requireTryonToken, async (req, res) => {
  const response = await fetch(`${HP_BASE}/api/uploads/user-image`, {
    method: "POST",
    headers: { "X-API-Key": process.env.HP_API_KEY },
    body: req,
  });
  res.json(await response.json());
});
```

```js Next.js API Routes
// /pages/api/tryon/submit.js
import jwt from "jsonwebtoken";
export const config = { api: { bodyParser: false } };

export default async function handler(req, res) {
  const token = (req.headers.authorization || "").replace("Bearer ", "");
  try { jwt.verify(token, process.env.JWT_SECRET); }
  catch { return res.status(401).json({ error: "Invalid token" }); }

  const response = await fetch(
    "https://hyperpersona-api-914497143112.us-central1.run.app/api/tryon/submit",
    {
      method: "POST",
      headers: {
        "X-API-Key": process.env.HP_API_KEY,
        "Content-Type": req.headers["content-type"],
      },
      body: req,
      duplex: "half",
    }
  );
  const data = await response.json();
  res.json(data);
}

// /pages/api/tryon/status/[jobId].js
import jwt from "jsonwebtoken";

export default async function handler(req, res) {
  const token = (req.headers.authorization || "").replace("Bearer ", "");
  try { jwt.verify(token, process.env.JWT_SECRET); }
  catch { return res.status(401).json({ error: "Invalid token" }); }

  const { jobId } = req.query;
  const response = await fetch(
    `https://hyperpersona-api-914497143112.us-central1.run.app/api/tryon/status/${jobId}`,
    { headers: { "X-API-Key": process.env.HP_API_KEY } }
  );

  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");

  const reader = response.body.getReader();
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    res.write(value);
  }
  res.end();
}
```

## Step 3 — Frontend Setup

```ts
import { Personalize } from "@gennoctua/personalize-core";

const sdk = await Personalize.init({
  auth: {
    proxyUrl: "/api/tryon",
    getToken: async () => {
      const res = await fetch("/auth/tryon-token", { method: "POST" });
      const data = await res.json();
      return data.token;
    }
  },
  analytics: {
    onEvent: (event) => console.log("[tryon]", event.eventName)
  }
});
```

```ts
document.getElementById("photoInput").addEventListener("change", async (e) => {
  const statusEl = document.getElementById("uploadStatus");
  statusEl.textContent = "Processing photos...";

  const summary = await sdk.ingestImages(e.target.files, (progress) => {
    statusEl.textContent = progress.message;
  });

  if (summary.availableCategories.length > 0) {
    statusEl.textContent = `Ready! ${summary.totalSelected} profile(s) selected.`;
    document.getElementById("tryOnBtn").disabled = false;
  } else {
    statusEl.textContent = "No usable photos found. Please upload clear face photos.";
  }
});
```

document.getElementById("tryOnBtn").addEventListener("click", async () => {
  const statusEl = document.getElementById("tryOnStatus");
  const resultEl = document.getElementById("resultImage");

  statusEl.textContent = "Generating your try-on...";

  try {
    const result = await sdk.personalize({
      imageUrl: "https://yourbrand.com/product.jpg",
      productType: "sunglasses",
      gender: "female",
      productId: "sku-123"
    });

    resultEl.src = result.imageUrl;
    resultEl.style.display = "block";
    statusEl.textContent = result.cacheHit ? "Loaded from cache!" : "Done!";
  } catch (err) {
    statusEl.textContent = "Try-on failed. Please try again.";
  }
});

## Common Issues

<AccordionGroup>
  <Accordion title='"Invalid token" error'>
    Check your JWT_SECRET is set in `.env`. Make sure `getToken` is returning the token string, not the full response object.
  </Accordion>
  <Accordion title='"Failed to fetch" error'>
    Check your proxy endpoints exist and are running. Make sure CORS is configured on your backend.
  </Accordion>
  <Accordion title='"Personalization timed out"'>
    HP may be under load — retry. Increase `pollMaxAttempts` in config if needed.
  </Accordion>
  <Accordion title="Next.js Turbopack issue">
    Use `next dev --webpack` instead of `next dev`.
  </Accordion>
</AccordionGroup>
